35 int PDelay::cpuNUM()
36 {
37 #ifdef WIN32
38  SYSTEM_INFO sysinfo;
39  GetSystemInfo(&sysinfo);
40  return sysinfo.dwNumberOfProcessors;
41 
42 #elif MACOS
43  int nm[2];
44  size_t len = 4;
45  uint32_t count;
46 
47  nm[0] = CTL_HW; nm[1] = HW_AVAILCPU;
48  sysctl(nm, 2, &count, &len, NULL, 0);
49 
50  if(count < 1) {
51  nm[1] = HW_NCPU;
52  sysctl(nm, 2, &count, &len, NULL, 0);
53  if(count < 1) { count = 1; }
54  }
55  return count;
56 #else
57  return sysconf(_SC_NPROCESSORS_ONLN);
58 #endif
59 }
