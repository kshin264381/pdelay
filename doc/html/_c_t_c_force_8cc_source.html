18 // Returns thermal velocity
19 //
20 // Vth = sqrt(3*k_B*T/m*)
21 //
22 Vel CTCForce::thermal_vel(const spCarrier& carrier)
23 {
24  // The thermal velocity comes from...
25  //
26  // E_therm = 3/2 * k_B * temperature
27  // E_kinetic = 1/2 * mass * velocity^2
28  //
29  // Since E_therm == E_kinetic....
30  //
31  // vel_therm = sqrt(3.0 * k_B * temperature / mass) (m/s)
32  //
33  fp_t vel_therm = \
34  v_therm(this->temperature, carrier->GetMass());
35 
36  // Implementing it into a random 3D vector.
37  Vel output_Vel = UnitVec3D<fp_t>() * vel_therm;
38  // note that the return value is in the unit of (m/s)
39  return output_Vel;
40 }

71 {
72  auto Mu = FP_T(0.0);
73 
74  if (carrier->GetTypeI() == CARR_T_HOLE)
75  Mu = this->DetMaterial.GetSemi("Silicon", "MUP");
76  else if (carrier->GetTypeI() == CARR_T_ELECTRON)
77  Mu = this->DetMaterial.GetSemi("Silicon", "MUN");
78 
79  auto Dt = k_B*this->temperature*Mu/q_h;

81 
82  auto DiffusedPos = UnitVec3D<fp_t>() * DiffLen;
83 
84  carrier->AdjPosDelta(DiffusedPos);

96 // Adjust position of a carrier from Brownian motion.
97 // (with given time)
98 void CTCForce::MFPAdj(spCarrier& carrier, const fp_t& tau)
99 {
100  // Obtain 3D thermal velocity.
101  //
102  fp_t mean_free_time = uniform_rand<fp_t>(1e-14, 1e-13);
103  //
104  // thermal velocity * meal_free_time --> random movement.
105  //
106  // Simulates brownian motion within this->delta_t period.
107  //
108  int brownian_motions = \
109  static_cast<int>(
110  round(tau / mean_free_time));
111  if (brownian_motions > 0) {
112  for (int i = 0; i<brownian_motions; ++i) {
113  carrier->AdjPosDelta(
114  this->thermal_vel(carrier)*this->len_scale_f*mean_free_time);
115  }
116  }
117 
118  return;
119 }

150 // Coulomb force calculation (returns MKS)
151 Force CTCForce::CoulombForce(spCarrier& carrier, spCarrier& other)
152 {
153  // Safeguard... if carrier == other... then just return dummy.
154  if (carrier->GetPos() == other->GetPos())
155  return ZeroForce;
156 
157  // Distance between current carrier and other (as mks unit.)
158  // Note that the carrier position is stored in (um) so we are
159  // converting it to MKS unit.
160  fp_t distance = \
161  carrier->GetPos().dist(other->GetPos()) / this->len_scale_f;
162 
163  // Handles debye length
164  // --> Coulomb force is not relevant if the distance between
165  // carriers are closer than Debye Length. (MKS)
166  fp_t debye_length = this->DebyeLength(carrier);
167  if ( fp_lteq<fp_t>(distance, debye_length) )
168  return ZeroForce;
169 
170  // Calculating force from the Carrier 'other'
171  // Note that the distance is in um, so adding up 1e-6
172  // multiplication factor to result MKS unit: (N).
173  fp_t force = \
174  k_e*carrier->GetCharge()*other->GetCharge()/(distance*distance);
175 
176  // Normalized direction vector.
177  Force f_direction = \
178  carrier->GetPos().direction(other->GetPos());
179 
180  return f_direction * static_cast<fp_t>(force);
181 }
