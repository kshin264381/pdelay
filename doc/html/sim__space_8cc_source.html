139 //
140 bool sim_space::is_collectable(spCarrier& carrier, Loc prev_pos)
141 {
142  //
143  // In fact, just detecting whether the carrier is within the
144  // scope of X - Y range is not enough to determine
145  // carrier collection at the electrodes.
146  // Rather, we need to estimate whether the carrier actually
147  // traveled through the electrode or not.
148  //
149  auto curr_pos = carrier->GetPos();
150 
151  auto plane_start = \
152  XY{ this->silicon_dimension->x_start,
153  this->silicon_dimension->x_end };
154  auto plane_end = \
155  XY{ this->silicon_dimension->y_start,
156  this->silicon_dimension->y_end };
157  auto anode_loc = this->silicon_dimension->z_start;
158  auto cathode_loc = this->silicon_dimension->z_end;
159  std::string plane_axis("xy");
160 
161  // Determine if the carrier is close to anode or cathode.
162  auto distA = fabs(curr_pos.z - anode_loc);
163  auto distC = fabs(curr_pos.z - cathode_loc);
164 
165  auto pass_anode = fp_chk_cross_3D<fp_t>(
166  prev_pos,
167  curr_pos, plane_axis, plane_start,
168  plane_end, anode_loc);
169  auto pass_cathode = fp_chk_cross_3D<fp_t>(
170  prev_pos,
171  curr_pos, plane_axis, plane_start,
172  plane_end, cathode_loc);
173 
174  if (pass_anode || pass_cathode) {
175  return true;
176  }
177  else if (pass_anode && pass_cathode) {
178  std::cerr \
179  << "Passing both anode and cathode?" \
180  << std::endl;
181  return false;
182  }
183  // Still stays inside... except nan situation.
184  else if (!(pass_anode && pass_cathode)) {
185  return false;
186  }
187  else {
188  return false;
189  }
190 }
