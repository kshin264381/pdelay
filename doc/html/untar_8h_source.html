43 static uint64_t decodeTarOctal(char* data, size_t size = 12)
44 {
45  unsigned char* currentPtr = (unsigned char*)data + size;
46  uint64_t sum = 0;
47  uint64_t currentMultiplier = 1;
48 
49  // Skipping everything after NUL/space character.
50  unsigned char* checkPtr = currentPtr;
51  for (; checkPtr >= (unsigned char*)data; checkPtr--) {
52  if ((*checkPtr) == 0 || (*checkPtr) == ' ') {
53  currentPtr = --checkPtr;
