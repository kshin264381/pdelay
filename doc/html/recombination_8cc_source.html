46 fp_t Recombination::simple(

49 {
50  /* Intrinsic case: simply no net recombination rate */
51  cfp_t n_i = this->n_i;
52  if ( fp_lteq(n_conc, n_i) && this->is_ntype(type) )
53  return static_cast<fp_t>(0.0);
54  if ( fp_lteq(p_conc, n_i) && this->is_ptype(type) )
55  return static_cast<fp_t>(0.0);
56 
57  if ( this->is_ntype(type) ) {
58  auto tau = \
59  this->MaterialData.GetSemi(this->material, "TAUN0");
60  return (n_conc - n_i)/tau;
61  }
62  else if ( this->is_ptype(type) ) {
63  auto tau = \
64  this->MaterialData.GetSemi(this->material, "TAUP0");
65  return (p_conc - n_i)/tau;
66  }
67

69 }

81 {

95 fp_t Recombination::ctrap(cfp_t& n_conc, cfp_t& p_conc, cfp_t& dop_conc, cfp_t& e_trap)

99  fp_t taup0 = this->MaterialData.GetSemi(this->material, "TAUP0");
100  fp_t taun0 = this->MaterialData.GetSemi(this->material, "TAUN0");

102  fp_t bn = this->MaterialData.GetSemi(this->material, "BN");
103  fp_t cn = this->MaterialData.GetSemi(this->material, "CN");
104  fp_t en = this->MaterialData.GetSemi(this->material, "EN");
105  fp_t ap = this->MaterialData.GetSemi(this->material, "AP");
106  fp_t bp = this->MaterialData.GetSemi(this->material, "BP");
107  fp_t cp = this->MaterialData.GetSemi(this->material, "CP");
108  fp_t ep = this->MaterialData.GetSemi(this->material, "EP");
109  fp_t nsrhn = this->MaterialData.GetSemi(this->material, "NSRHN");
110  fp_t nsrhp = this->MaterialData.GetSemi(this->material, "NSRHP");
111 
112  fp_t taun = taun0/(an+bn*(dop_conc/nsrhn) + cn*pow(dop_conc/nsrhn, en));
113  fp_t taup = taup0/(ap+bp*(dop_conc/nsrhp) + cp*pow(dop_conc/nsrhp, ep));
114 
115  fp_t num = \
116  taup*(n_conc+this->n_i*exp(e_trap/(k_B_eV*this->temperature))) + \
117  taun*(p_conc+this->n_i*exp(-1.0*e_trap/(k_B_eV*this->temperature)));
118
